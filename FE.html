        class RAGInterface {

            constructor() {

                this.documents = [];

                this.isProcessing = false;

                this.vectorStore = null;

                this.initializeElements();

            }


            initializeElements() {

                this.fileUpload = document.getElementById('fileUpload');

                this.fileInput = document.getElementById('fileInput');

                this.uploadBtn = document.getElementById('uploadBtn');

                this.systemStatus = document.getElementById('systemStatus');

                this.progressBar = document.getElementById('progressBar');

                this.progressFill = document.getElementById('progressFill');

                this.documentsList = document.getElementById('documentsList');

                this.chatMessages = document.getElementById('chatMessages');

                this.questionInput = document.getElementById('questionInput');

                this.sendBtn = document.getElementById('sendBtn');

            }


            bindEvents() {

                // File upload events

                this.fileUpload.addEventListener('click', () => this.fileInput.click());

                this.fileInput.addEventListener('change', this.handleFileSelect.bind(this));

                this.uploadBtn.addEventListener('click', this.uploadDocuments.bind(this));


                // Drag and drop events

                this.fileUpload.addEventListener('dragover', this.handleDragOver.bind(this));

                this.fileUpload.addEventListener('dragleave', this.handleDragLeave.bind(this));

                this.fileUpload.addEventListener('drop', this.handleDrop.bind(this));


                // Chat events

                this.sendBtn.addEventListener('click', this.sendMessage.bind(this));

                this.questionInput.addEventListener('keypress', (e) => {

                    if (e.key === 'Enter' && !e.shiftKey) {

                        e.preventDefault();

                        this.sendMessage();

                    }

                });

            }


            handleDragOver(e) {

                e.preventDefault();

                this.fileUpload.classList.add('dragover');

            }


            handleDragLeave(e) {

                e.preventDefault();

                this.fileUpload.classList.remove('dragover');

            }


            handleDrop(e) {

                e.preventDefault();

                this.fileUpload.classList.remove('dragover');

                const files = Array.from(e.dataTransfer.files);

                this.fileInput.files = e.dataTransfer.files;

                this.handleFileSelect({ target: { files } });

            }


            handleFileSelect(e) {

                const files = Array.from(e.target.files);

                const validFiles = files.filter(file => {

                    const validTypes = ['.pdf', '.txt', '.docx', '.md', '.doc'];

                    return validTypes.some(type => file.name.toLowerCase().endsWith(type));

                });


                if (validFiles.length !== files.length) {

                    this.addMessage('system', 'Some files were filtered out. Only PDF, TXT, DOCX, and MD files are supported.');

                }


                this.updateUploadButton(validFiles.length > 0);

            }


            updateUploadButton(hasFiles) {

                this.uploadBtn.disabled = !hasFiles || this.isProcessing;

                this.uploadBtn.textContent = hasFiles ? 

                    `Upload ${this.fileInput.files.length} file(s)` : 'Upload Documents';

            }


            async uploadDocuments() {

                const files = Array.from(this.fileInput.files);

                if (files.length === 0) return;


                this.isProcessing = true;

                this.updateStatus('processing', '⏳ Processing documents...');

                this.showProgress();

                

                try {

                    // Simulate document processing

                    for (let i = 0; i < files.length; i++) {

                        const file = files[i];

                        await this.processDocument(file);

                        this.updateProgress((i + 1) / files.length * 100);

                    }


                    // Simulate embedding creation

                    this.updateStatus('processing', '🧠 Creating embeddings...');

                    await this.sleep(2000);


                    // Add documents to store

                    files.forEach(file => {

                        this.documents.push({

                            name: file.name,

                            size: file.size,

                            processed: true

                        });

                    });


                    this.updateDocumentsList();

                    this.updateStatus('ready', '✅ Documents processed and ready!');

                    this.enableChat();

                    this.addMessage('system', `Successfully processed ${files.length} document(s). You can now ask questions!`);


                } catch (error) {

                    this.updateStatus('error', '❌ Error processing documents');

                    this.addMessage('system', 'Error processing documents. Please try again.');

                } finally {

                    this.isProcessing = false;

                    this.hideProgress();

                    this.fileInput.value = '';

                    this.updateUploadButton(false);

                }

            }


            async processDocument(file) {

                // Simulate document parsing based on file type

                const extension = file.name.split('.').pop().toLowerCase();

                let processingTime = 1000; // Base processing time


                switch (extension) {

                    case 'pdf':

                        processingTime = 2000;

                        break;

                    case 'docx':

                        processingTime = 1500;

                        break;

                    case 'txt':

                    case 'md':

                        processingTime = 500;

                        break;

                }


                await this.sleep(processingTime);

            }


            async sendMessage() {

                const question = this.questionInput.value.trim();

                if (!question || this.isProcessing) return;


                if (this.documents.length === 0) {

                    this.addMessage('system', 'Please upload documents first before asking questions.');

                    return;

                }


                this.addMessage('user', question);

                this.questionInput.value = '';

                this.sendBtn.disabled = true;

                this.sendBtn.innerHTML = '<div class="loading"></div>';


                try {

                    // Simulate RAG process

                    const response = await this.performRAG(question);

                    this.addMessage('assistant', response);

                } catch (error) {

                    this.addMessage('assistant', 'Sorry, I encountered an error while processing your question.');

                } finally {

                    this.sendBtn.disabled = false;

                    this.sendBtn.textContent = 'Send';

                }

            }


            async performRAG(question) {

                // Simulate retrieval and generation

                await this.sleep(1500);


                // Simple keyword matching simulation for demo

                const keywords = question.toLowerCase().split(' ');

                const relevantDocs = this.documents.filter(doc => 

                    keywords.some(keyword => doc.name.toLowerCase().includes(keyword))

                );


                // Simulate relevance check

                const hasRelevantInfo = Math.random() > 0.3; // 70% chance of finding relevant info


                if (!hasRelevantInfo) {

                    return "Nothing relevant found.";

                }


                // Generate a mock response based on the question

                const responses = [

                    `Based on the uploaded documents, I found relevant information about "${question}". The documents contain detailed insights that address your query.`,

                    `According to the information in your documents, here's what I found regarding "${question}": The content provides comprehensive coverage of this topic.`,

                    `From analyzing your uploaded documents, I can provide information about "${question}". The relevant sections contain valuable details that answer your question.`

                ];


                return responses[Math.floor(Math.random() * responses.length)];

            }


            addMessage(type, content) {

                const message = document.createElement('div');

                message.className = `message ${type}`;

                message.textContent = content;

                this.chatMessages.appendChild(message);

                this.chatMessages.scrollTop = this.chatMessages.scrollHeight;

            }


            updateStatus(type, message) {

                this.systemStatus.className = `status ${type}`;

                this.systemStatus.textContent = message;

            }


            showProgress() {

                this.progressBar.style.display = 'block';

                this.progressFill.style.width = '0%';

            }


            hideProgress() {

                this.progressBar.style.display = 'none';

            }


            updateProgress(percentage) {

                this.progressFill.style.width = `${percentage}%`;

            }


            updateDocumentsList() {

                if (this.documents.length === 0) {

                    this.documentsList.innerHTML = `

                        <p style="color: #64748b; text-align: center; padding: 20px;">

                            No documents uploaded yet

                        </p>

                    `;

                    return;

                }


                this.documentsList.innerHTML = this.documents.map((doc, index) => `

                    <div class="document-item">

                        <span class="document-name">${doc.name}</span>

                        <button class="remove-btn" onclick="ragInterface.removeDocument(${index})">

                            Remove

                        </button>

                    </div>

                `).join('');

            }


            removeDocument(index) {

                this.documents.splice(index, 1);

                this.updateDocumentsList();

                

                if (this.documents.length === 0) {

                    this.disableChat();

                    this.updateStatus('ready', '✅ System Ready');

                    this.addMessage('system', 'All documents removed. Please upload new documents to continue.');

                }

            }


            enableChat() {

                this.questionInput.disabled = false;

                this.sendBtn.disabled = false;

                this.questionInput.placeholder = "Ask a question about your documents...";

            }


            disableChat() {

                this.questionInput.disabled = true;

                this.sendBtn.disabled = true;

                this.questionInput.placeholder = "Upload documents first...";

            }


            sleep(ms) {

                return new Promise(resolve => setTimeout(resolve, ms));

            }

        }
